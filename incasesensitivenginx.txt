Incase Sensitive nginx URL

/etc/nginx/nginx.conf

http{

 ## Incase sensitive
        perl_set $uri_lowercase 'sub {
        my $r = shift;
        my $uri = $r->uri;
        $uri = lc($uri);
        return $uri;
        }';

}


# /etc/nginx/sites-enabled/yoursite.conf

server {
    
    # ...

    location ~ [A-Z] {
        rewrite ^(.*)$ $scheme://$host$uri_lowercase;
    }    
    
    # ...
    
}
